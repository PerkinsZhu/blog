```
docker search rabbitmq

docker pull rabbitmq

docker run -itd --rm --name rabbitmq -e RABBITMQ_DEFAULT_USER=admin -e RABBITMQ_DEFAULT_PASS=admin -p 15672:15672 -p 5672:5672 rabbitmq

// 15672进入rabbitmq数据管理后台的

// 5672是通过代码接口访问队列的


// 宿主机器安装目录
cd /etc/rabbitmq
```
